# Work-in-progress

expr             -> term termrest

termrest         -> '+' termrest
                  | '-' termrest
                  | {e}

term             -> subterm subtermrest

subtermrest      -> '*' subtermrest
                  | '/' subtermrest
                  | {e}

subterm          -> factor factorrest

factorrest       -> '^' factorrest
                  | {e}

factor           -> optionalNegation number | '(' expr ')'

functionName     -> 'cos' | 'sin' | 'tan' | 'log' | 'ln' | 'sqrt'
number           -> digits optionalFraction optionalExponent | 'e' | 'pi'
optionalNegation -> - | {e}
optionalFraction -> '.' digits | {e}
optionalExponent -> ( ( 'E' | 'e' ) ( '+' | '-' | {e} ) digits ) | {e}
digits           -> digit digit*
digit            -> [0-9]
